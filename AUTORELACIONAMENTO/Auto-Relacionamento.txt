							Auto-Relacionamento

create table cursos(
idcurso int primary key auto_increment,
nome varchar(30),
horas int,
valor float(10,2),
id_prereq int
);

alter table cursos add constraint FK_PREREQ
FOREIGN KEY(id_prereq) references CURSOS(idcurso);

Traga o NOME, VALOR, HORAS E O NOME DO PRE REQUISITO DO CURSO */
select 	c.nome as curso,
		c.valor,
		c.horas as carga,
		IFNULL(p.nome,'---') as 'Pré-Requisitos'    <- IF o valor for nulo trocar por '---' e mudar o nome..
from cursos c  										<- Tabela curso C
left join cursos p  								<- Tabela Pré-requisito P
on p.idcurso = c.id_prereq;
